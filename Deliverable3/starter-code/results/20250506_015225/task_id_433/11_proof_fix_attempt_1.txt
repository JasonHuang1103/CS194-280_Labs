induction a using Array.inductionOn with
| nil =>
  simp at *
| cons hd tl ih =>
  unfold isGreater isGreater_spec
  simp only [Array.foldl, List.mem] at *
  split
  路 intro h i hi
    cases i
    路 apply h
      simp
    路 apply ih
      exact fun j hj => h (j + 1) (Nat.succ_lt_succ hj)
  路 intro h
    apply ih
    intro j hj
    apply h
    exact Nat.succ_lt_succ hj
