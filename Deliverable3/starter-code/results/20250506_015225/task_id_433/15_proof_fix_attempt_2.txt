let l := a.toList 
induction l with
| nil => simp 
| cons hd tl ih =>
  simp only [List.foldl, List.foldr] at *
  split
  路 intro h i hi
    cases i
    路 exact h 0 (by simp)
    路 apply ih
      intros j hj
      apply h
      exact succ_lt_succ hj
  路 intro h
    apply ih
    intro j hj
    apply h
    exact succ_lt_succ hj
