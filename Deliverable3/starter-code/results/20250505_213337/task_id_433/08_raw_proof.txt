induction a using Array.inductionOn with
| nil =>
    simp
| cons x xs ih =>
    simp [Array.foldl, ih]
    constructor
    路 intro h i hi
      cases hi with
      | head => exact h.1
      | tail hi' => exact h.2 i hi'
    路 intro h
      constructor
      路 exact h 0 (Nat.zero_lt_succ _)
      路 intro i hi
        exact h (i + 1) (Nat.succ_lt_succ hi)
